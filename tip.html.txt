    function shuffle(array) {
      return array.sort(() => Math.random() - 0.5);
    }

    function showTrivia() {
      const t = trivia[Math.floor(Math.random() * trivia.length)];
      const shuffled = shuffle([...t.options]);
      let html = `<h3>${t.question}</h3>`;
      shuffled.forEach(opt => {
        html += `<button class='btn-option' onclick='checkTrivia("${opt}", "${t.correct}")'>${opt}</button>`;
      });
      document.getElementById("gameBox").innerHTML = html;
    }

    function checkTrivia(answer, correct) {
      let message = answer === correct
        ? "Correct! You're a real fan!"
        : `Oops! The correct answer was: <strong>${correct}</strong>`;
      document.getElementById("gameBox").innerHTML = `<p>${message}</p>`;
      showTipBox();
    }

        <button class='btn' onclick='checkRiddle()'>Submit</button>
      `;
      document.getElementById("gameBox").innerHTML = html;
    }

    function checkRiddle() {
