  <script>
    const questions = [
      {
        question: "Complete the line: 'huenda life pande yako ni tricky na ganji ukipata haimalizi?'",
        correct: "wiki",
        options: ["mwezi", "shida", "mwaka", "wiki"]
      },
      {
        question: "Which is the last song dropped by Brix Stem71?",
        correct: "street",
        options: ["live", "ghetto yut", "enemies", "gwap", "street"]
      },
      {
        question: "Guess the next song that Brix Stem will drop?",
        correct: "possible",
        options: ["last year", "my girl", "zogo", "possible"]
      },
      {
        question: "Guess the area that Stem represents?",
        correct: "sabamoja",
        options: ["buru", "kayole", "dandora", "sabamoja"]
      }
    ];

    const picked = questions[Math.floor(Math.random() * questions.length)];
    document.getElementById('questionText').textContent = picked.question;

    const optionsContainer = document.getElementById('options');
    picked.options.forEach(option => {
      const btn = document.createElement('button');
      btn.textContent = option;
      btn.onclick = () => {
        const result = document.getElementById('result');
        if (option === picked.correct) {
          result.textContent = "Correct! You're a real fan.";
        } else {
          result.textContent = `Oops! The correct answer was "${picked.correct}".`;
        }
        document.getElementById('tipSection').style.display = 'block';
        optionsContainer.querySelectorAll('button').forEach(b => b.disabled = true);
      };
      optionsContainer.appendChild(btn);
    });
  </script>
